31"""A simple wrapper around enum types to expose utility functions. 

32 

33Instances are created as properties with the same name as the enum they wrap 

34on proto classes. For usage, see: 

35 reflection_test.py 

36""" 

37 

38__author__ = 'rabsatt@google.com (Kevin Rabsatt)' 

39 

40import six 

41 

42 

43class EnumTypeWrapper(object): 

44 """A utility for finding the names of enum values.""" 

45 

46 DESCRIPTOR = None 

47 

48 def __init__(self, enum_type): 

49 """Inits EnumTypeWrapper with an EnumDescriptor.""" 

50 self._enum_type = enum_type 

51 self.DESCRIPTOR = enum_type 

52 

53 def Name(self, number): 

54 """Returns a string containing the name of an enum value.""" 

55 if number in self._enum_type.values_by_number: 

56 return self._enum_type.values_by_number[number].name 

57 

58 if not isinstance(number, six.integer_types): 

59 raise TypeError('Enum value for %s must be an int, but got %r.' % ( 

60 self._enum_type.name, number)) 

61 else: 

62 # %r here to handle the odd case when you pass in a boolean. 

63 raise ValueError('Enum %s has no name defined for value %r' % ( 

64 self._enum_type.name, number)) 

65 

66 def Value(self, name): 

67 """Returns the value corresponding to the given enum name.""" 

68 if name in self._enum_type.values_by_name: 

69 return self._enum_type.values_by_name[name].number 

70 raise ValueError('Enum %s has no value defined for name %s' % ( 

71 self._enum_type.name, name)) 

72 

73 def keys(self): 

74 """Return a list of the string names in the enum. 

75 

76 These are returned in the order they were defined in the .proto file. 

77 """ 

78 

79 return [value_descriptor.name 

80 for value_descriptor in self._enum_type.values] 

81 

82 def values(self): 

83 """Return a list of the integer values in the enum. 

84 

85 These are returned in the order they were defined in the .proto file. 

86 """ 

87 

88 return [value_descriptor.number 

89 for value_descriptor in self._enum_type.values] 

90 

91 def items(self): 

92 """Return a list of the (name, value) pairs of the enum. 

93 

94 These are returned in the order they were defined in the .proto file. 

95 """ 

96 return [(value_descriptor.name, value_descriptor.number) 

97 for value_descriptor in self._enum_type.values] 

98 

99 def __getattr__(self, name): 

100 """Returns the value corresponding to the given enum name.""" 

101 if name in self._enum_type.values_by_name: 

102 return self._enum_type.values_by_name[name].number 

103 raise AttributeError

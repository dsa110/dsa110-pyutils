26@six.add_metaclass(abc.ABCMeta) 

27class wait_base(object): 

28 """Abstract base class for wait strategies.""" 

29 

30 @abc.abstractmethod 

31 def __call__(self, retry_state): 

32 pass 

33 

34 def __add__(self, other): 

35 return wait_combine(self, other) 

36 

37 def __radd__(self, other): 

38 # make it possible to use multiple waits with the built-in sum function 

39 if other == 0: 

40 return self 

41 return self.__add__(other)

62class wait_random(wait_base): 

63 """Wait strategy that waits a random amount of time between min/max.""" 

64 

65 def __init__(self, min=0, max=1): # noqa 

66 self.wait_random_min = min 

67 self.wait_random_max = max 

68 

69 @_compat.wait_dunder_call_accept_old_params 

70 def __call__(self, retry_state): 

71 return (self.wait_random_min +

76class wait_combine(wait_base): 

77 """Combine several waiting strategies.""" 

78 

79 def __init__(self, *strategies): 

80 self.wait_funcs = tuple(_compat.wait_func_accept_retry_state(strategy)

88class wait_chain(wait_base): 

89 """Chain two or more waiting strategies. 

90 

91 If all strategies are exhausted, the very last strategy is used 

92 thereafter. 

93 

94 For example:: 

95 

96 @retry(wait=wait_chain(*[wait_fixed(1) for i in range(3)] + 

97 [wait_fixed(2) for j in range(5)] + 

98 [wait_fixed(5) for k in range(4))) 

99 def wait_chained(): 

100 print("Wait 1s for 3 attempts, 2s for 5 attempts and 5s 

101 thereafter.") 

102 """ 

103 

104 def __init__(self, *strategies): 

105 self.strategies = [_compat.wait_func_accept_retry_state(strategy) 

106 for strategy in strategies] 

107 

108 @_compat.wait_dunder_call_accept_old_params 

109 def __call__(self, retry_state): 

110 wait_func_no = min(max(retry_state.attempt_number, 1), 

111 len(self.strategies)) 

112 wait_func = self.strategies[wait_func_no - 1] 

113 return wait_func(retry_state=retry_state)

116class wait_incrementing(wait_base): 

117 """Wait an incremental amount of time after each attempt. 

118 

119 Starting at a starting value and incrementing by a value for each attempt 

120 (and restricting the upper limit to some maximum value). 

121 """ 

122 

123 def __init__(self, start=0, increment=100, max=_utils.MAX_WAIT): # noqa 

124 self.start = start 

125 self.increment = increment 

126 self.max = max 

127 

128 @_compat.wait_dunder_call_accept_old_params 

129 def __call__(self, retry_state): 

130 result = self.start + ( 

131 self.increment * (retry_state.attempt_number - 1) 

132 ) 

133 return max(0, min(result, self.max))

136class wait_exponential(wait_base): 

137 """Wait strategy that applies exponential backoff. 

138 

139 It allows for a customized multiplier and an ability to restrict the 

140 upper and lower limits to some maximum and minimum value. 

141 

142 The intervals are fixed (i.e. there is no jitter), so this strategy is 

143 suitable for balancing retries against latency when a required resource is 

144 unavailable for an unknown duration, but *not* suitable for resolving 

145 contention between multiple processes for a shared resource. Use 

146 wait_random_exponential for the latter case. 

147 """ 

148 

149 def __init__(self, multiplier=1, max=_utils.MAX_WAIT, exp_base=2, min=0): # noqa 

150 self.multiplier = multiplier 

151 self.min = min 

152 self.max = max 

153 self.exp_base = exp_base 

154 

155 @_compat.wait_dunder_call_accept_old_params 

156 def __call__(self, retry_state): 

157 try: 

158 exp = self.exp_base ** (retry_state.attempt_number - 1) 

159 result = self.multiplier * exp 

160 except OverflowError: 

161 return self.max 

162 return max(max(0, self.min), min(result, self.max))

165class wait_random_exponential(wait_exponential): 

166 """Random wait with exponentially widening window. 

167 

168 An exponential backoff strategy used to mediate contention between multiple 

169 uncoordinated processes for a shared resource in distributed systems. This 

170 is the sense in which "exponential backoff" is meant in e.g. Ethernet 

171 networking, and corresponds to the "Full Jitter" algorithm described in 

172 this blog post: 

173 

174 https://aws.amazon.com/blogs/architecture/exponential-backoff-and-jitter/ 

175 

176 Each retry occurs at a random time in a geometrically expanding interval. 

177 It allows for a custom multiplier and an ability to restrict the upper 

178 limit of the random interval to some maximum value. 

179 

180 Example:: 

181 

182 wait_random_exponential(multiplier=0.5, # initial window 0.5s 

183 max=60) # max 60s timeout 

184 

185 When waiting for an unavailable resource to become available again, as 

186 opposed to trying to resolve contention for a shared resource, the 

187 wait_exponential strategy (which uses a fixed interval) may be preferable. 

188 

189 """ 

190 

191 @_compat.wait_dunder_call_accept_old_params 

192 def __call__(self, retry_state): 

193 high = super(wait_random_exponential, self).__call__( 

194 retry_state=retry_state) 

195 return random.uniform(0, high)

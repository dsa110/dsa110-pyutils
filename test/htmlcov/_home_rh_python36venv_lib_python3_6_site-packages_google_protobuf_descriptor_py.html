138 def GetOptions(self): 

139 """Retrieves descriptor options. 

140 

141 This method returns the options set or creates the default options for the 

142 descriptor. 

143 """

145 return self._options

393 

394 """Descriptor for a single field in a .proto file. 

395 

396 A FieldDescriptor instance has the following attributes: 

397 

398 name: (str) Name of this field, exactly as it appears in .proto. 

399 full_name: (str) Name of this field, including containing scope. This is 

400 particularly relevant for extensions. 

401 camelcase_name: (str) Camelcase name of this field. 

402 index: (int) Dense, 0-indexed index giving the order that this 

403 field textually appears within its message in the .proto file. 

404 number: (int) Tag number declared for this field in the .proto file. 

405 

406 type: (One of the TYPE_* constants below) Declared type. 

407 cpp_type: (One of the CPPTYPE_* constants below) C++ type used to 

408 represent this field. 

409 

410 label: (One of the LABEL_* constants below) Tells whether this 

411 field is optional, required, or repeated. 

412 has_default_value: (bool) True if this field has a default value defined, 

413 otherwise false. 

414 default_value: (Varies) Default value of this field. Only 

415 meaningful for non-repeated scalar fields. Repeated fields 

416 should always set this to [], and non-repeated composite 

417 fields should always set this to None. 

418 

419 containing_type: (Descriptor) Descriptor of the protocol message 

420 type that contains this field. Set by the Descriptor constructor 

421 if we're passed into one. 

422 Somewhat confusingly, for extension fields, this is the 

423 descriptor of the EXTENDED message, not the descriptor 

424 of the message containing this field. (See is_extension and 

425 extension_scope below). 

426 message_type: (Descriptor) If a composite field, a descriptor 

427 of the message type contained in this field. Otherwise, this is None. 

428 enum_type: (EnumDescriptor) If this field contains an enum, a 

429 descriptor of that enum. Otherwise, this is None. 

430 

431 is_extension: True iff this describes an extension field. 

432 extension_scope: (Descriptor) Only meaningful if is_extension is True. 

433 Gives the message that immediately contains this extension field. 

434 Will be None iff we're a top-level (file-level) extension field. 

435 

436 options: (descriptor_pb2.FieldOptions) Protocol message field options or 

437 None to use default field options. 

438

931def _ToCamelCase(name): 

932 """Converts name to camel-case and returns it.""" 

933 capitalize_next = False 

934 result = [] 

935 

936 for c in name: 

937 if c == '_': 

938 if result: 

939 capitalize_next = True 

940 elif capitalize_next: 

941 result.append(c.upper()) 

942 capitalize_next = False 

943 else: 

944 result += c 

945 

946 # Lower-case the first letter. 

947 if result and result[0].isupper(): 

948 result[0] = result[0].lower() 

949 return ''.join(result)

977def MakeDescriptor(desc_proto, package='', build_file_if_cpp=True, 

978 syntax=None): 

979 """Make a protobuf Descriptor given a DescriptorProto protobuf. 

980 

981 Handles nested descriptors. Note that this is limited to the scope of defining 

982 a message inside of another message. Composite fields can currently only be 

983 resolved if the message is defined in the same scope as the field. 

984 

985 Args: 

986 desc_proto: The descriptor_pb2.DescriptorProto protobuf message. 

987 package: Optional package name for the new message Descriptor (string). 

988 build_file_if_cpp: Update the C++ descriptor pool if api matches. 

989 Set to False on recursion, so no duplicates are created. 

990 syntax: The syntax/semantics that should be used. Set to "proto3" to get 

991 proto3 field presence semantics. 

992 Returns: 

993 A Descriptor for protobuf messages. 

994 """ 

995 if api_implementation.Type() == 'cpp' and build_file_if_cpp: 

996 # The C++ implementation requires all descriptors to be backed by the same 

997 # definition in the C++ descriptor pool. To do this, we build a 

998 # FileDescriptorProto with the same definition as this descriptor and build 

999 # it into the pool. 

1000 from google.protobuf import descriptor_pb2 

1001 file_descriptor_proto = descriptor_pb2.FileDescriptorProto() 

1002 file_descriptor_proto.message_type.add().MergeFrom(desc_proto) 

1003 

1004 # Generate a random name for this proto file to prevent conflicts with any 

1005 # imported ones. We need to specify a file name so the descriptor pool 

1006 # accepts our FileDescriptorProto, but it is not important what that file 

1007 # name is actually set to. 

1008 proto_name = binascii.hexlify(os.urandom(16)).decode('ascii') 

1009 

1010 if package: 

1011 file_descriptor_proto.name = os.path.join(package.replace('.', '/'), 

1012 proto_name + '.proto') 

1013 file_descriptor_proto.package = package 

1014 else: 

1015 file_descriptor_proto.name = proto_name + '.proto' 

1016 

1017 _message.default_pool.Add(file_descriptor_proto) 

1018 result = _message.default_pool.FindFileByName(file_descriptor_proto.name) 

1019 

1020 if _USE_C_DESCRIPTORS: 

1021 return result.message_types_by_name[desc_proto.name] 

1022 

1023 full_message_name = [desc_proto.name] 

1024 if package: full_message_name.insert(0, package) 

1025 

1026 # Create Descriptors for enum types 

1027 enum_types = {} 

1028 for enum_proto in desc_proto.enum_type: 

1029 full_name = '.'.join(full_message_name + [enum_proto.name]) 

1030 enum_desc = EnumDescriptor( 

1031 enum_proto.name, full_name, None, [ 

1032 EnumValueDescriptor(enum_val.name, ii, enum_val.number) 

1033 for ii, enum_val in enumerate(enum_proto.value)]) 

1034 enum_types[full_name] = enum_desc 

1035 

1036 # Create Descriptors for nested types 

1037 nested_types = {} 

1038 for nested_proto in desc_proto.nested_type: 

1039 full_name = '.'.join(full_message_name + [nested_proto.name]) 

1040 # Nested types are just those defined inside of the message, not all types 

1041 # used by fields in the message, so no loops are possible here. 

1042 nested_desc = MakeDescriptor(nested_proto, 

1043 package='.'.join(full_message_name), 

1044 build_file_if_cpp=False, 

1045 syntax=syntax) 

1046 nested_types[full_name] = nested_desc 

1047 

1048 fields = [] 

1049 for field_proto in desc_proto.field: 

1050 full_name = '.'.join(full_message_name + [field_proto.name]) 

1051 enum_desc = None 

1052 nested_desc = None 

1053 if field_proto.json_name: 

1054 json_name = field_proto.json_name 

1055 else: 

1056 json_name = None 

1057 if field_proto.HasField('type_name'): 

1058 type_name = field_proto.type_name 

1059 full_type_name = '.'.join(full_message_name + 

1060 [type_name[type_name.rfind('.')+1:]]) 

1061 if full_type_name in nested_types: 

1062 nested_desc = nested_types[full_type_name] 

1063 elif full_type_name in enum_types: 

1064 enum_desc = enum_types[full_type_name] 

1065 # Else type_name references a non-local type, which isn't implemented 

1066 field = FieldDescriptor( 

1067 field_proto.name, full_name, field_proto.number - 1, 

1068 field_proto.number, field_proto.type, 

1069 FieldDescriptor.ProtoTypeToCppProtoType(field_proto.type), 

1070 field_proto.label, None, nested_desc, enum_desc, None, False, None, 

1071 options=_OptionsOrNone(field_proto), has_default_value=False, 

1072 json_name=json_name) 

1073 fields.append(field) 

1074 

1075 desc_name = '.'.join(full_message_name) 

1076 return Descriptor(desc_proto.name, desc_name, None, None, fields, 

1077 list(nested_types.values()), list(enum_types.values()), [], 

1078 options=_OptionsOrNone(desc_proto))
